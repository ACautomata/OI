# 快速读写
>如果想尝试一下比较可的优化，大概“寥寥”60几行吧，所以还是老实的用`read()`
```C++
int read()
{
	int x=0,w=1;
	char ch=0;
	while (ch<'0'||ch>'9')    
	{
		if (ch=='-') w=-1;     
		ch=getchar();             
	}
	while (ch>='0'&&ch<='9')   
	{
		x=x*10+(ch-'0');  
		ch=getchar(); 
	}
	return x*w;
}
inline void write(int x)
{
	static int sta[35];
	int top=0;
	do
		sta[top++]=x%10,x/=10;
	while (x);
	while (top) putchar(sta[--top]+48);
}
```

# LCA
链式前向星
```C++
inline void add(int u,int v)
{
    edge[++cnt].next=head[u];
    edge[cnt].to=v;
    head[u]=cnt;
}
```
第一个dfs用来初始化每一个结点的深度及其父节点
```C++
inline void dfs(int u,int fa)
{
	f[u][0]=fa;
	de[u]=de[fa]+1;
	
	for(int i=1; (1<<i)<=de[u]; i++)
		f[u][i]=f[f[u][i-1]][i-1]; 
		
	for(int i=head[u]; i; i=edge[i].next)
	{
		int v=edge[i].to;
		if(v!=fa) 
			dfs(v,u);
	}
}
```
然后就是LCA的过程了
```C++
inline int lca(int u,int v)
{
	if(depth[u]<depth[v])
		swap(u,v);
	for(int i=20; i>=0; i--)
	{
		if((1<<i)<=depth[u]-depth[v])
			u=f[u][i];
	}
	if(u==v) 
		return u;
	for(int i=29; i>=0; i--)
		if(f[u][i]!=f[v][i])
		{
			u=f[u][i];
			v=f[v][i];
		}
	return f[u][0];
}
```
第二次dfs进行简单的统计，每个结点累加其所有孩子结点的权值，再取最大值
```C++
inline void ask(int u,int fa)
{
    for(int i=head[u];i;i=edge[i].next)
    {
        int v=edge[i].to;
        if(v!=fa)
        {
            aks(v,u);
            sum[u]+=sum[v];
        }
    }
    ans=max(ans,sum[u]);
}
```
